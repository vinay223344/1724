#!/usr/bin/env python3
import sys
import os
import re
import shutil

# -------------------------------------------------
# 1. Check Command Line Argument
# -------------------------------------------------
if len(sys.argv) != 2:
    print("Required number of command line arguments are not provided!")
    sys.exit(1)

verilog_file = sys.argv[1]

# -------------------------------------------------
# 2. Check if Verilog file exists
# -------------------------------------------------
if not os.path.exists(verilog_file):
    print("The provided Verilog file does not exist!")
    sys.exit(1)

# -------------------------------------------------
# 3. Create / Handle WORKDIR
# -------------------------------------------------
workdir = "./WORKDIR"

if os.path.exists(workdir):
    ans = input("WORKDIR already exists. Overwrite? (Y/N): ")
    if ans.lower() != 'y':
        print("Exiting script.")
        sys.exit(0)
    else:
        shutil.rmtree(workdir)

os.makedirs(workdir)

# -------------------------------------------------
# 4. Read File and Remove Comments
# -------------------------------------------------
with open(verilog_file, 'r') as f:
    code = f.read()

# remove multiline comments
code = re.sub(r'/\*.*?\*/', '', code, flags=re.S)

# remove single line comments
code = re.sub(r'//.*', '', code)

lines = code.split('\n')

# -------------------------------------------------
# 5. Initialize Variables
# -------------------------------------------------
inputs = []
outputs = []
inouts = []
ports = set()

instances = {}
compile_errors = []

top_module = None

# -------------------------------------------------
# 6. Extract Top Module Name
# -------------------------------------------------
for i, line in enumerate(lines):
    m = re.search(r'\bmodule\s+(\w+)', line)
    if m:
        top_module = m.group(1)
        break

if not top_module:
    compile_errors.append("ERR03: missing module declaration")

# -------------------------------------------------
# 7. Parse Ports
# -------------------------------------------------
port_pattern = r'(input|output|inout)\s*(\[[^\]]+\])?\s*([^;]+);'

for i, line in enumerate(lines):

    # Case sensitivity check
    if re.search(r'\bInput\b|\bOutput\b|\bInout\b', line):
        compile_errors.append(f"ERR09: case sensitivity issue of port direction in line {i+1}")

    m = re.search(port_pattern, line)

    if m:
        direction = m.group(1)
        width = m.group(2)
        names = m.group(3)

        if width is None:
            width = "1"

        names = names.split(',')

        for name in names:
            name = name.strip()

            if not re.match(r'^[A-Za-z_]\w*$', name):
                compile_errors.append(f"ERR04: illegal port name {name} in line {i+1}")
                continue

            if name in ports:
                compile_errors.append(f"ERR05: duplicate port name {name} in line {i+1}")
                continue

            ports.add(name)

            if direction == "input":
                inputs.append((name, width))
            elif direction == "output":
                outputs.append((name, width))
            else:
                inouts.append((name, width))

    # Missing semicolon
    if re.search(r'(input|output|inout).*[^;]$', line) and not line.strip().endswith(';'):
        compile_errors.append(f"ERR01: missing semicolon ';' in line {i+1}")

# -------------------------------------------------
# 8. Detect Submodule Instantiations
# -------------------------------------------------
inst_pattern = r'^\s*(\w+)\s+(\w+)\s*\('

for i, line in enumerate(lines):

    m = re.search(inst_pattern, line)

    if m:
        cell = m.group(1)
        inst = m.group(2)

        if not re.match(r'^[A-Za-z_]\w*$', inst):
            compile_errors.append(f"ERR04: illegal instance name {inst} in line {i+1}")
            continue

        if inst in instances:
            compile_errors.append(f"ERR11: Instance name {inst} is already defined in line {i+1}")
        else:
            instances[inst] = cell

# -------------------------------------------------
# 9. Parenthesis Check
# -------------------------------------------------
for i, line in enumerate(lines):
    if line.count('(') != line.count(')'):
        compile_errors.append(f"ERR02: missing parenthesis in line {i+1}")

# -------------------------------------------------
# 10. begin-end / module-endmodule check
# -------------------------------------------------
begin_count = code.count("begin")
end_count = code.count("end")

module_count = code.count("module")
endmodule_count = code.count("endmodule")

if begin_count != end_count or module_count != endmodule_count:
    compile_errors.append("ERR03: missing endmodule/end for the module/always block")

# -------------------------------------------------
# 11. Create Directory Structure
# -------------------------------------------------
top_dir = os.path.join(workdir, top_module)
os.makedirs(top_dir)

input_file = open(os.path.join(top_dir, "inputs"), "w")
output_file = open(os.path.join(top_dir, "outputs"), "w")
inout_file = open(os.path.join(top_dir, "inouts"), "w")
sub_file = open(os.path.join(top_dir, "submodules"), "w")

# -------------------------------------------------
# 12. Write Ports
# -------------------------------------------------
for name, width in inputs:
    input_file.write(f"{name}, {width}\n")

for name, width in outputs:
    output_file.write(f"{name}, {width}\n")

for name, width in inouts:
    inout_file.write(f"{name}, {width}\n")

# -------------------------------------------------
# 13. Write Submodules
# -------------------------------------------------
for inst, cell in instances.items():
    sub_file.write(f"{cell}, {inst}\n")

input_file.close()
output_file.close()
inout_file.close()
sub_file.close()

# -------------------------------------------------
# 14. Write compile.log
# -------------------------------------------------
log_file = open(os.path.join(workdir, "compile.log"), "w")

for err in compile_errors:
    log_file.write(err + "\n")

log_file.close()

print("Parsing completed.")
print("Check WORKDIR for outputs.")
