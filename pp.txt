import os
import re
import shutil
import sys

def strip_verilog_comments(code):
    code = re.sub(r"//.*", "", code) # removing single and multiline comments
    code = re.sub(r"/\*.*?\*/", "", code, flags=re.S)
    return code

def detect_verilog_errors(verilog_lines, log_file):
    error_list = []
    stack_module, stack_begin = [], []
    found_ports, found_instances = set(), set()

#The enumerate() function in Python is a built-in function , which produces pairs of an index and the corresponding element from the iterable during iteration.

    for idx, line in enumerate(verilog_lines, 1):  #semicolon checking
        clean = line.strip()
 
        if re.search(r"(input|output|inout|wire|reg|assign|OPAMP|res|cap)", line) and not clean.endswith(";"):
            error_list.append(f"ERR01: missing semicolon at line {idx}")

        if line.count("(") != line.count(")"): #parantheseis balance checking
            error_list.append(f"ERR02: mismatched parenthesis at line {idx}")

        if re.search(r"\bmodule\b", line): # module and endmodule checking
            stack_module.append(idx)
        if re.search(r"\bendmodule\b", line) and stack_module:
            stack_module.pop()

        if re.search(r"\bbegin\b", line): # begin,end checking
            stack_begin.append(idx)
        if re.search(r"\bend\b", line) and stack_begin:
            stack_begin.pop()


        if re.search(r"\b(input|output|inout)\b", line): # checking for  illegal port name
            tokens = line.strip().split()
            for tok in tokens:
                if re.match(r"^[0-9]", tok) or re.search(r"[-!@#$%^&*]", tok):
                    error_list.append(f"ERR04: illegal port name {tok} in line {idx}")

        m = re.search(r"\b(input|output|inout)\b\s*(?:\[[^]]+\])?\s*([\w_]+)", line) # checking for dupliacte port definiton
        if m:
            port = m.group(2)
            if port in found_ports:
                error_list.append(f"ERR05: duplicate port {port} at line {idx}")
            else:
                found_ports.add(port)

        if re.match(r"^\s*\[.*\]\s*\w+", line) and not re.search(r"\b(input|output|inout)\b", line):#missing direction for the port in line
            error_list.append(f"ERR06: missing port direction at line {idx}")

        if re.search(r"module\s+\w+\s*\([^)]*,$", clean): # comma misuses
            error_list.append(f"ERR07: trailing comma in portlist at line {idx}")

        # Multiple width specifiers
        if re.search(r"(input|output|inout)\s+\[[^]]+\]\s+\[[^]]+\]", line):
            error_list.append(f"ERR08: multiple widths declared on same port in line {idx}")

        # Case-sensitive keywords
        if re.search(r"\b(Input|Output|Inout)\b", line):
            error_list.append(f"ERR09: port direction case error at line {idx}")

        # Line-break bug in portlist
        if re.search(r"module\s+\w+\s*\(([^)]*)\)", line):
            plist = re.search(r"\(([^)]*)\)", line).group(1)
            if re.search(r"\w+\s*\n\s*\w+", plist):
                error_list.append(f"ERR10: missing comma in portlist in line {idx}")

        # Instance name reuse
        instance_match = re.search(r"(\w+)\s+(\w+)\s*\(", line)
        if instance_match:
            inst = instance_match.group(2)
            if inst in found_instances:
                error_list.append(f"ERR11: duplicate instance {inst} at line {idx}")
            else:
                found_instances.add(inst)

    # Unmatched module/begin
    for m_idx in stack_module:
        error_list.append(f"ERR03: module without endmodule at line {m_idx}")
    for b_idx in stack_begin:
        error_list.append(f"ERR03: begin without end at line {b_idx}")

    # Output discovered errors
    with open(log_file, "w") as log:
        for err in error_list:
            log.write(err + "\n")

def epai(code): # extract_ports_and_instances
    port_dict = {"input": [], "output": [], "inout": []}
    top_module, submods = None, []

    module_match = re.search(r"module\s+(\w+)", code)    #For  Finding  top module

    if module_match:
        top_module = module_match.group(1)

    ports = re.findall(r"(input|output|inout)\s*(\[[^]]+\])?\s*([^;]+);", code) # For Finding input/output/inout ports

    for direction, width, names in ports:
        width = width or "1"
        for name in map(str.strip, names.split(',')):
            if name:
                port_dict[direction].append((name, width))

    
    for match in re.findall(r"(\w+)\s+(\w+)\s*\([^;]*\);", code): #for detecting unique submodules ans their instances
        submods.append(match)
    return top_module, port_dict, submods

def output_files(top_module, portinfo, submods, result_dir):
    os.mkdir(result_dir)
    # Write port files
    for port_type in portinfo:
        with open(os.path.join(result_dir, port_type + "s"), "w") as f:
            for p, w in portinfo[port_type]:
                f.write(f"{p}, {w}\n")
    # Submodules
    unique_subinsts = set()
    with open(os.path.join(result_dir, "submodules"), "w") as f:
        for cell, instance in submods:
            if instance not in unique_subinsts:
                unique_subinsts.add(instance)
                f.write(f"{cell}, {instance}\n")
