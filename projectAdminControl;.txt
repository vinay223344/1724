#!/usr/bin/env python3.6

import sys
import os
import shutil
import getpass


if len(sys.argv) < 4:
    print("Error: Required 3 arguments not provided")
    sys.exit()

if len(sys.argv) > 4:
    print("Warning: Only first 3 arguments will be used")

src = sys.argv[1]
dest = sys.argv[2]
config = sys.argv[3]


if not os.path.exists(src) or not os.path.isdir(src):
    print("Error: Source directory invalid")
    sys.exit()


if not os.path.exists(config) or not os.path.isfile(config):
    print("Error: Config file invalid")
    sys.exit()


if os.path.exists(dest):
    ch = input("Destination exists. Overwrite? (Y/N): ")
    if ch.lower() != 'y':
        sys.exit()
    shutil.rmtree(dest)

os.makedirs(dest)


soft = []
suffix = []
noneed = []
mode = None

with open(config) as f:
    for line in f:
        line = line.strip()

        if not line or line.startswith("#"):
            continue

        if line.startswith("SOFT_LINK"):
            mode = "soft"
            continue
        elif line.startswith("SUFFIX"):
            mode = "suffix"
            continue
        elif line.startswith("NO_NEED"):
            mode = "noneed"
            continue

        if mode == "soft":
            soft.append(line)
        elif mode == "suffix":
            suffix.append(line)
        elif mode == "noneed":
            noneed.append(line)


username = getpass.getuser()

def process(src_path, dest_path):

    for item in os.listdir(src_path):

        s = os.path.join(src_path, item)
        d = os.path.join(dest_path, item)

        # Skip NO_NEED
        if s in noneed:
            continue

        # SOFT LINK
        if s in soft:
            if not os.path.exists(d):
                os.symlink(s, d)
            continue

        # SUFFIX RULE
        if s in suffix:
            name = item + "_" + username
            d = os.path.join(dest_path, name)

        if os.path.isdir(s):

            if not os.path.exists(d):
                os.makedirs(d)

            process(s, d)

        else:
            if not os.path.exists(d):
                shutil.copy2(s, d)

process(src, dest)

print("Project directory created successfully")
