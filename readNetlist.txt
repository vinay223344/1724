import sys
import os
import shutil
import re

if len(sys.argv) < 4:
    print("Error: Required 3 command line arguments not provided.")
    print("Usage: python3.6 readNetlist.py <spectreNetlist.scs> <netlistHierarchy> <topCellName>")
    sys.exit(1)

if len(sys.argv) > 4:
    print("Warning: More than 3 arguments provided. Only first 3 will be considered.")

netlist_file = sys.argv[1]
root_dir = sys.argv[2]
top_cell = sys.argv[3]



if not os.path.exists(netlist_file):
    print("Error: Spectre netlist file does not exist.")
    sys.exit(1)

if not os.path.isfile(netlist_file):
    print("Error: Given netlist path is not a file.")
    sys.exit(1)



if os.path.exists(root_dir):
    choice = input(f"Directory '{root_dir}' already exists. Overwrite? (Y/N): ")
    if choice.lower() != 'y':
        print("Exiting without overwriting directory.")
        sys.exit(0)
    shutil.rmtree(root_dir)

os.makedirs(root_dir)



def parse_netlist(file_path):
    subckt_dict = {}
    current_subckt = None
    buffer = []

    with open(file_path, 'r') as f:
        for line in f:
            stripped = line.strip()

            if not stripped or stripped.startswith('//'):
                continue

            if stripped.lower().startswith("subckt"):
                parts = stripped.split()
                current_subckt = parts[1]
                buffer = []
                continue

            if stripped.lower().startswith("ends"):
                if current_subckt:
                    subckt_dict[current_subckt] = buffer
                current_subckt = None
                continue

            if current_subckt:
                buffer.append(stripped)

    return subckt_dict


subckts = parse_netlist(netlist_file)

if top_cell not in subckts:
    print(f"Error: Top cell '{top_cell}' not found in netlist.")
    sys.exit(1)


def extract_mosfet(line):
    tokens = line.split()
    instance_name = tokens[0]
    model_name = tokens[2] if len(tokens) > 2 else ""

    width = re.search(r"w=([\d\.]+)", line)
    length = re.search(r"l=([\d\.]+)", line)
    m = re.search(r"m=\((\d+)\)\*\((\d+)\)", line)

    width_val = width.group(1) if width else ""
    length_val = length.group(1) if length else ""
    multiplier = m.group(1) if m else ""
    fingers = m.group(2) if m else ""

    return {
        "instance": instance_name,
        "model": model_name,
        "width": width_val,
        "length": length_val,
        "multiplier": multiplier,
        "fingers": fingers
    }


def process_cell(cell_name, parent_path):
    cell_path = os.path.join(parent_path, cell_name)
    os.makedirs(cell_path, exist_ok=True)

    leaf_file_path = os.path.join(cell_path, "leafCells.txt")
    leaf_entries = []

    lines = subckts.get(cell_name, [])

    for line in lines:
        if re.match(r"^I\w+", line):
            parts = line.split()
            inst_name = parts[0]
            sub_cell = parts[-1]

            dir_name = f"{inst_name}_{sub_cell}"
            inst_path = os.path.join(cell_path, dir_name)

            process_cell(sub_cell, cell_path)

        elif re.match(r"^M\w+", line):
            mos = extract_mosfet(line)
            leaf_entries.append(mos)

    if leaf_entries:
        with open(leaf_file_path, 'w') as f:
            for mos in leaf_entries:
                f.write(f"Instance Name : {mos['instance']}\n")
                f.write(f"Model Name    : {mos['model']}\n")
                f.write(f"Width (um)    : {mos['width']}\n")
                f.write(f"Length (um)   : {mos['length']}\n")
                f.write(f"Multiplier    : {mos['multiplier']}\n")
                f.write(f"Fingers       : {mos['fingers']}\n")
                f.write("-" * 40 + "\n")


process_cell(top_cell, root_dir)

print("Hierarchy generation completed successfully.")
